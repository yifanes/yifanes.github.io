#函数作为参数的例子-数据和操作隔离

```go
package main

import (
	"fmt"
	"strings"
)

func main() {
	list := []string{
		"go scanner",
		"go parser",
		"go compiler",
		"go printer",
		"go formatter",
	}
	chain := []func(string) string{
		remotePrefix,
		strings.TrimSpace,
		strings.ToUpper,
	}

	StringProcess(list, chain)

	for _, str := range list {
		fmt.Println(str)
	}
}

func remotePrefix(str string) string {
	return strings.TrimPrefix(str, "go")
}

func StringProcess(list []string, chain []func(string) string) {
	for index, str := range list {
		result := str

		for _, proc := range chain {
			result = proc(result)
		}
		list[index] = result
	}
}
```
